    for sim in simulations[mode]:
        input_file, frames, replacement_mode, debug_mode = sim

        results = run_simulation(input_file, frames, replacement_mode, debug_mode)

import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd

data = []

for mode in total_results:
    if total_results[mode]:
        for sim, fault in zip(simulations[mode], total_results[mode]):
            data.append({'Frames': sim[1], 'Page Hit': fault, 'Mode': mode})

df = pd.DataFrame(data)

sns.set(style="whitegrid")

plt.figure(figsize=(15, 8))
sns.lineplot(x='Frames', y='Page Hit', hue='Mode', data=df)

plt.xlabel('Frames')
plt.ylabel('Hit Rate')
plt.title('Hit Rate vs Frames for Each Replacement Mode (Trace File: bzip.trace)')

frame_max = df['Frames'].max()
plt.xticks(range(0, frame_max + 1, 4))  # range(0, max_frame, step=4)

plt.show()
